// Ejemplo básico de producto punto entre matrices

// Matriz 2×3
A = [[1, 2, 3], 
     [4, 5, 6]];

// Matriz 3×2
B = [[7, 8], 
     [9, 10], 
     [11, 12]];

// Producto punto: (2×3) · (3×2) = (2×2)
C = A · B;

// Matriz 2×1
D = [[1], 
     [2]];

// Producto encadenado: (2×2) · (2×1) = (2×1)
E = C · D;

// Usando función punto
F = A.punto(B);

// Con paréntesis
G = (A · B) · D;

// Matriz identidad 2×2
I = [[1, 0], 
     [0, 1]];

// Producto con identidad
H = I · C;
